43 /**
44  * Sends an internal control event to each core set in 'mask'.
45  *
46  * When all cores set in 'mask' has processed the event an additional
47  * internal 'done' msg is sent to synchronize - this done-event can then
48  * trigger any notifications for the user that the operation was completed.
49  * Processing of the 'done' event will also call the 'f_done_callback'
50  * function if given.
51  *
52  * @return 0 on success, otherwise return the number of ctrl events that could
53  * not be sent due to error
54  */
55 int
56 send_core_ctrl_events(const em_core_mask_t *const mask, em_event_t ctrl_event,
57  void (*f_done_callback)(void *arg_ptr),
58  void *f_done_arg_ptr,
59  int num_notif, const em_notif_t notif_tbl[])
60 {
61  em_status_t err;
62  em_event_group_t event_group = EM_EVENT_GROUP_UNDEF;
63  const internal_event_t *i_event = em_event_pointer(ctrl_event);
64  const int core_count = em_core_count(); /* All running EM cores */
65  const int mask_count = em_core_mask_count(mask); /* Subset of cores*/
66  int alloc_count = 0;
67  int sent_count = 0;
68  int unsent_count = mask_count;
69  int first_qidx;
70  int i;
71  em_event_t events[mask_count];
72 
73  if (unlikely(num_notif > EM_EVENT_GROUP_MAX_NOTIF)) {
74  INTERNAL_ERROR(EM_ERR_TOO_LARGE, EM_ESCOPE_INTERNAL_NOTIF,
75  "Too large notif table (%i)", num_notif);
76  return unsent_count;
77  }
78 
79  event_group = em_event_group_create();
80  if (unlikely(event_group == EM_EVENT_GROUP_UNDEF)) {
81  INTERNAL_ERROR(EM_ERR_ALLOC_FAILED, EM_ESCOPE_INTERNAL_NOTIF,
82  "Event group alloc failed");
83  return unsent_count;
84  }
85 
86  /*
87  * Set up internal notification when all cores are done.
88  */
89  err = internal_done_w_notif_req(event_group, mask_count,
90  f_done_callback, f_done_arg_ptr,
91  num_notif, notif_tbl);
92  if (unlikely(err != EM_OK)) {
93  INTERNAL_ERROR(err, EM_ESCOPE_INTERNAL_NOTIF,
94  "Internal 'done' notif setup failed");
95  goto err_delete_event_group;
96  }
97 
98  /*
99  * Allocate ctrl events to be sent to the concerned cores.
100  * Reuse the input ctrl_event later so alloc one less.
101  * Copy content from input ctrl_event into all allocated events.
102  */
103  for (i = 0; i < mask_count - 1; i++) {
104  events[i] = em_alloc(sizeof(internal_event_t),
105  EM_EVENT_TYPE_SW,
106  EM_POOL_DEFAULT);
107  if (unlikely(events[i] == EM_EVENT_UNDEF)) {
108  INTERNAL_ERROR(EM_ERR_ALLOC_FAILED,
109  EM_ESCOPE_INTERNAL_NOTIF,
110  "Internal event alloc failed");
111  goto err_free_events;
112  }
113  alloc_count++;
114 
115  internal_event_t *i_event_tmp = em_event_pointer(events[i]);
116  /* Copy input event content */
117  *i_event_tmp = *i_event;
118  }
119  /* Reuse the input event */
120  events[i] = ctrl_event;
121  /* don't increment alloc_count++, caller frees input event on error */
122 
123  /*
124  * Send ctrl events to the concerned cores
125  */
126  first_qidx = queue_id2idx(FIRST_INTERNAL_QUEUE);
127 
128  for (i = 0; i < core_count; i++) {
129  if (em_core_mask_isset(i, mask)) {
130  /*
131  * Send copy to each core-specific queue,
132  * track completion using an event group.
133  */
134  err = em_send_group(events[sent_count],
135  queue_idx2hdl(first_qidx + i),
136  event_group);
137  if (unlikely(err != EM_OK)) {
138  INTERNAL_ERROR(err, EM_ESCOPE_INTERNAL_NOTIF,
139  "Event group send failed");
140  goto err_free_events;
141  }
142  sent_count++;
143  unsent_count--;
144  }
145  }
146 
147  return 0; /* Success, all ctrl events sent */
148 
149  /* Error handling, free resources */
150 err_free_events:
151  for (i = sent_count; i < alloc_count; i++)
152  em_free(events[i]);
153 err_delete_event_group:
154  (void)em_event_group_abort(event_group);
155  (void)em_event_group_delete(event_group);
156  return unsent_count;
157 }

159 /**
160  * Receive function for handling internal ctrl events, called by the daemon EO
161  */
162 void
163 internal_event_receive(void *eo_ctx, em_event_t event, em_event_type_t type,
164  em_queue_t queue, void *q_ctx)
165 {
166  const internal_event_t *i_event;
167 
168  /* currently unused args */
169  (void)eo_ctx;
170  (void)type;
171  (void)q_ctx;
172 
173  i_event = em_event_pointer(event);
174 
175  switch (i_event->id) {
176  /*
177  * Internal Done event
178  */
179  case EM_INTERNAL_DONE:
180  i_event__internal_done(i_event);
181  break;
182  /*
183  * Internal event related to Queue Group modification: remove a core
184  */
185  case QUEUE_GROUP_REM_REQ:
186  case QUEUE_GROUP_REM_SYNC_REQ:
187  /*
188  * Nothing needs to be done, enough to ensure the core is not
189  * processing an event from a queue in the modified queue group
190  */
191  break;
192  /*
193  * Internal events related to EO local start&stop functionality
194  */
195  case EO_START_LOCAL_REQ:
196  case EO_START_SYNC_LOCAL_REQ:
197  case EO_STOP_LOCAL_REQ:
198  case EO_STOP_SYNC_LOCAL_REQ:
199  case EO_REM_QUEUE_LOCAL_REQ:
200  case EO_REM_QUEUE_SYNC_LOCAL_REQ:
201  case EO_REM_QUEUE_ALL_LOCAL_REQ:
202  case EO_REM_QUEUE_ALL_SYNC_LOCAL_REQ:
203  i_event__eo_local_func_call_req(i_event);
204  break;
205 
206  default:
207  INTERNAL_ERROR(EM_ERR_BAD_ID,
208  EM_ESCOPE_INTERNAL_EVENT_RECV_FUNC,
209  "Internal ev-id:0x%" PRIx64 " Q:%" PRI_QUEUE "",
210  i_event->id, queue);
211  break;
212  }
213 
214  em_free(event);
215 }

217 /**
218  * Handle the internal 'done' event
219  */
220 static void
221 i_event__internal_done(const internal_event_t *i_ev)
222 {
223  int num_notif;
224  em_status_t ret;
225 
226  /* Release the event group, we are done with it */
227  ret = em_event_group_delete(i_ev->done.event_group);
228 
229  if (unlikely(ret != EM_OK))
230  INTERNAL_ERROR(ret, EM_ESCOPE_EVENT_INTERNAL_DONE,
231  "Event group %" PRI_EGRP " delete failed (ret=%u)",
232  i_ev->done.event_group, ret);
233 
234  /* Call the callback function, performs custom actions at 'done' */
235  if (i_ev->done.f_done_callback != NULL)
236  i_ev->done.f_done_callback(i_ev->done.f_done_arg_ptr);
237 
238  /*
239  * Send notification events if requested by the caller.
240  */
241  num_notif = i_ev->done.num_notif;
242 
243  if (num_notif > 0) {
244  ret = send_notifs(num_notif, i_ev->done.notif_tbl);
245  if (unlikely(ret != EM_OK))
246  INTERNAL_ERROR(ret, EM_ESCOPE_EVENT_INTERNAL_DONE,
247  "em_send() of notifs(%d) failed",
248  num_notif);
249  }
250 }

252 /**
253  * Helper func: Allocate & set up the internal 'done' event with
254  * function callbacks and notification events.
255  */
256 em_status_t
257 internal_done_w_notif_req(em_event_group_t event_group,
258  int event_group_count,
259  void (*f_done_callback)(void *arg_ptr),
260  void *f_done_arg_ptr,
261  int num_notif, const em_notif_t notif_tbl[])
262 {
263  em_event_t event;
264  internal_event_t *i_event;
265  em_notif_t i_notif;
266  em_status_t err;
267  int i;
268 
269  event = em_alloc(sizeof(internal_event_t), EM_EVENT_TYPE_SW,
270  EM_POOL_DEFAULT);
271  RETURN_ERROR_IF(event == EM_EVENT_UNDEF, EM_ERR_ALLOC_FAILED,
272  EM_ESCOPE_INTERNAL_DONE_W_NOTIF_REQ,
273  "Internal event 'DONE' alloc failed!");
274 
275  i_event = em_event_pointer(event);
276  i_event->id = EM_INTERNAL_DONE;
277  i_event->done.event_group = event_group;
278  i_event->done.f_done_callback = f_done_callback;
279  i_event->done.f_done_arg_ptr = f_done_arg_ptr;
280  i_event->done.num_notif = num_notif;
281 
282  for (i = 0; i < num_notif; i++) {
283  i_event->done.notif_tbl[i].event = notif_tbl[i].event;
284  i_event->done.notif_tbl[i].queue = notif_tbl[i].queue;
285  i_event->done.notif_tbl[i].egroup = notif_tbl[i].egroup;
286  }
287 
288  i_notif.event = event;
289  i_notif.queue = queue_id2hdl(SHARED_INTERNAL_QUEUE);
290  i_notif.egroup = EM_EVENT_GROUP_UNDEF;
291 
292  /*
293  * Request sending of EM_INTERNAL_DONE when 'event_group_count' events
294  * in 'event_group' have been seen. The 'Done' event will trigger the
295  * notifications to be sent.
296  */
297  err = em_event_group_apply(event_group, event_group_count,
298  1, &i_notif);
299 
300  RETURN_ERROR_IF(err != EM_OK, err, EM_ESCOPE_INTERNAL_DONE_W_NOTIF_REQ,
301  "Event group apply failed");
302 
303  return EM_OK;
304 }

306 /**
307  * Helper func to send notifications events
308  */
309 em_status_t
310 send_notifs(const int num_notif, const em_notif_t notif_tbl[])
311 {
312  int i;
313  em_status_t err;
314 
315  for (i = 0; i < num_notif; i++) {
316  const em_event_t event = notif_tbl[i].event;
317  const em_queue_t queue = notif_tbl[i].queue;
318  const em_event_group_t egrp = notif_tbl[i].egroup;
319 
320  /* 'egroup' may be uninit in old appl code, check */
321  if (invalid_egrp(egrp))
322  err = em_send(event, queue);
323  else
324  err = em_send_group(event, queue, egrp);
325 
326  if (unlikely(err != EM_OK))
327  return err;
328  }
329 
330  return EM_OK;
331 }

333 em_status_t
334 check_notif(const em_notif_t *const notif)
335 {
336  if (unlikely(notif == NULL || notif->event == EM_EVENT_UNDEF))
337  return EM_ERR_BAD_POINTER;
338 
339  const bool is_external = queue_external(notif->queue);
340 
341  if (!is_external) {
342  const queue_elem_t *q_elem = queue_elem_get(notif->queue);
343 
344  if (unlikely(q_elem == NULL || !queue_allocated(q_elem)))
345  return EM_ERR_NOT_FOUND;
346  }
347 
348  if (notif->egroup != EM_EVENT_GROUP_UNDEF) {
349  const event_group_elem_t *egrp_elem =
350  event_group_elem_get(notif->egroup);
351 
352  if (unlikely(egrp_elem == NULL ||
353  !event_group_allocated(egrp_elem)))
354  return EM_ERR_BAD_ID;
355  }
356 
357  return EM_OK;
358 }

360 em_status_t
361 check_notif_tbl(const int num_notif, const em_notif_t notif_tbl[])
362 {
363  em_status_t err;
364  int i;
365 
366  if (unlikely((unsigned int)num_notif > EM_EVENT_GROUP_MAX_NOTIF))
367  return EM_ERR_TOO_LARGE;
368 
369  if (unlikely(num_notif > 0 && notif_tbl == NULL))
370  return EM_ERR_BAD_POINTER;
371 
372  for (i = 0; i < num_notif; i++) {
373  err = check_notif(&notif_tbl[i]);
374  if (unlikely(err != EM_OK))
375  return err;
376  }
377 
378  return EM_OK;
