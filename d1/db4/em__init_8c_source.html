10 poll_drain_mask_check(const em_core_mask_t *logic_mask,
11  const em_core_mask_t *poll_drain_mask)
12 {
13  /* check if mask is zero (all cores, OK) */
14  if (em_core_mask_iszero(poll_drain_mask))
15  return EM_OK;
16 
17  int i;
18 
19  /* check mask validity */
20  for (i = 0; i < EM_MAX_CORES; i++) {
21  if (em_core_mask_isset(i, poll_drain_mask) &&
22  !em_core_mask_isset(i, logic_mask))
23  return EM_ERR_OPERATION_FAILED;
24  }
25  return EM_OK;
26 }
