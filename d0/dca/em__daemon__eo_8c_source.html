44 void
45 daemon_eo_create(void)
46 {
47  em_eo_t eo;
48  em_status_t stat;
49  em_status_t stat_eo_start = EM_ERROR;
50 
51  eo = em_eo_create("daemon-eo", daemon_eo_start, NULL /*start_local*/,
52  daemon_eo_stop, NULL /*stop_local*/,
53  daemon_eo_receive, NULL);
54  if (eo == EM_EO_UNDEF)
55  DAEMON_ERROR(EM_FATAL(EM_ERR_BAD_ID), "daemon-eo create fail");
56 
57  /* Store the EO in shared memory */
58  em_shm->daemon.eo = eo;
59 
60  stat = em_eo_start(eo, &stat_eo_start, NULL, 0, NULL);
61  if (stat != EM_OK || stat_eo_start != EM_OK)
62  DAEMON_ERROR(EM_FATAL(EM_ERR_LIB_FAILED),
63  "daemon-eo start failed!");
64 }

66 void
67 daemon_eo_shutdown(void)
68 {
69  const int core = em_core_id();
70  const em_eo_t eo = em_shm->daemon.eo;
71  eo_elem_t *const eo_elem = eo_elem_get(eo);
72  em_status_t stat;
73 
74  EM_PRINT("%s() on EM-core %d\n", __func__, core);
75 
76  if (unlikely(eo_elem == NULL)) {
77  DAEMON_ERROR(EM_FATAL(EM_ERR_BAD_ID),
78  "daemon-eo handle:%" PRI_EO " invalid!", eo);
79  return;
80  }
81 
82  /*
83  * Stop the daemon-EO, i.e. call the daemon-EO global stop func.
84  * Note: cannot call normal API func em_eo_stop() since that would use
85  * internal ctrl events that might not be dispatched during shutdown.
86  */
87  /* Change state here to allow em_eo_delete() from EO stop func */
88  eo_elem->state = EM_EO_STATE_CREATED; /* == EO_STATE_STOPPED */
89  stat = eo_elem->stop_func(eo_elem->eo_ctx, eo);
90  if (stat != EM_OK)
91  DAEMON_ERROR(EM_FATAL(EM_ERR_LIB_FAILED),
92  "daemon-eo stop/delete failed!");
93 }

95 static em_status_t
96 daemon_eo_start(void *eo_ctx, em_eo_t eo, const em_eo_conf_t *conf)
97 {
98  const int num_cores = em_core_count();
99  char qgrp_name[] = EM_QUEUE_GROUP_CORE_LOCAL_BASE_NAME;
100  char q_name[EM_QUEUE_NAME_LEN] = "EM_Q_INTERNAL_000000";
101  em_queue_group_t queue_group;
102  em_queue_t queue;
103  em_queue_t shared_queue;
104  em_core_mask_t mask;
105  em_status_t stat;
106  const char *err_str = "";
107  int i;
108 
109  (void)eo_ctx;
110  (void)conf;
111 
112  EM_PRINT("daemon-eo starting!\n");
113 
114  /*
115  * Create static internal queue used for internal EM messaging.
116  * Cannot use em_queue_create_static() here since the requested handle
117  * 'SHARED_INTERNAL_QUEUE' lies outside of the normal static range.
118  */
119  shared_queue = queue_id2hdl(SHARED_INTERNAL_QUEUE);
120  queue = queue_create("EM_Q_INTERNAL_SHARED", EM_QUEUE_TYPE_ATOMIC,
121  INTERNAL_QUEUE_PRIORITY, EM_QUEUE_GROUP_DEFAULT,
122  shared_queue, EM_ATOMIC_GROUP_UNDEF,
123  NULL /* use default queue config */, &err_str);
124  if (queue == EM_QUEUE_UNDEF || queue != shared_queue)
125  return EM_FATAL(EM_ERR_NOT_FREE);
126 
127  stat = em_eo_add_queue(eo, shared_queue, 0, NULL);
128  if (stat != EM_OK)
129  return EM_FATAL(stat);
130 
131  for (i = 0; i < num_cores; i++) {
132  em_queue_t queue_req;
133 
134  em_core_mask_zero(&mask);
135  em_core_mask_set(i, &mask);
136  core_queue_grp_name(qgrp_name, i);
137 
138  /*
139  * Create per-core queue groups for core-specific queues
140  */
141  queue_group = em_queue_group_create(qgrp_name, &mask, 0, NULL);
142  if (unlikely(queue_group == EM_QUEUE_GROUP_UNDEF))
143  return EM_FATAL(EM_ERR_ALLOC_FAILED);
144 
145  queue_req = queue_id2hdl(FIRST_INTERNAL_QUEUE + i);
146  snprintf(&q_name[14], EM_QUEUE_NAME_LEN - 14,
147  "%" PRI_QUEUE "", queue_req);
148  q_name[EM_QUEUE_NAME_LEN - 1] = '\0';
149 
150  /*
151  * Create static internal per-core queues used for internal
152  * EM messaging. Cannot use em_queue_create_static() here since
153  * the requested handles lies outside of the normal static
154  * range.
155  */
156  queue = queue_create(q_name, EM_QUEUE_TYPE_ATOMIC,
157  INTERNAL_QUEUE_PRIORITY, queue_group,
158  queue_req, EM_ATOMIC_GROUP_UNDEF,
159  NULL, /* use default queue config */
160  &err_str);
161  if (unlikely(queue == EM_QUEUE_UNDEF || queue != queue_req))
162  return EM_FATAL(EM_ERR_NOT_FREE);
163 
164  stat = em_eo_add_queue(eo, queue, 0, NULL);
165  if (unlikely(stat != EM_OK))
166  return EM_FATAL(stat);
167  }
168 
169  return EM_OK;
170 }

172 static em_status_t
173 daemon_eo_stop(void *eo_ctx, em_eo_t eo)
174 {
175  em_status_t stat = EM_OK;
176  eo_elem_t *const eo_elem = eo_elem_get(eo);
177 
178  (void)eo_ctx;
179 
180  EM_PRINT("%s() on EM-core %d\n", __func__, em_core_id());
181 
182  if (unlikely(eo_elem == NULL)) {
183  stat = EM_FATAL(EM_ERR_BAD_ID);
184  DAEMON_ERROR(stat, "daemon-eo handle:%" PRI_EO " invalid!", eo);
185  return stat;
186  }
187 
188  /* Cannot use API funcs - internal ctrl events might not work */
189  stat = queue_disable_all(eo_elem);
190  stat |= eo_delete_queue_all(eo_elem);
191  /* Finally delete the daemon-eo, API func is ok here */
192  stat |= em_eo_delete(eo);
193 
194  return stat;
195 }
