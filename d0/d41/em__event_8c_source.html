78 void
79 output_queue_drain(const queue_elem_t *output_q_elem)
80 {
81  const em_queue_t output_queue = output_q_elem->queue;
82  const em_output_func_t output_fn =
83  output_q_elem->output.output_conf.output_fn;
84  void *const output_fn_args =
85  output_q_elem->output.output_conf.output_fn_args;
86 
87  const int deq_max = 32;
88 
89  em_event_t output_ev_tbl[deq_max];
90  /* use same event-tbl, dequeue odp events into the EM event-tbl */
91  odp_event_t *const odp_deq_events = (odp_event_t *)output_ev_tbl;
92 
93  const odp_queue_t odp_queue = output_q_elem->odp_queue;
94  unsigned int output_num;
95  int deq;
96  int ret;
97 
98  const bool esv_ena = esv_enabled();
99 
100  do {
101  deq = odp_queue_deq_multi(odp_queue,
102  odp_deq_events/*out=output_ev_tbl[]*/,
103  deq_max);
104  if (unlikely(deq <= 0))
105  return;
106 
107  output_num = (unsigned int)deq;
108  /* odp_deq_events[] == output_ev_tbl[], .evgen still missing */
109 
110  /* decrement pool statistics before passing events out-of-EM */
111  if (em_shm->opt.pool.statistics_enable || esv_ena) {
112  event_hdr_t *ev_hdrs[output_num];
113 
114  event_to_hdr_multi(output_ev_tbl, ev_hdrs, output_num);
115  if (em_shm->opt.pool.statistics_enable) {
116  poolstat_dec_evhdr_multi_output(ev_hdrs,
117  output_num);
118  }
119  if (esv_ena)
120  evstate_em2usr_multi(output_ev_tbl/*in/out*/,
121  ev_hdrs, output_num,
122  EVSTATE__OUTPUT_MULTI);
123  }
124 
125  ret = output_fn(output_ev_tbl, output_num,
126  output_queue, output_fn_args);
127 
128  if (unlikely((unsigned int)ret != output_num))
129  em_free_multi(&output_ev_tbl[ret], output_num - ret);
130  } while (deq > 0);
131 }

133 void
134 output_queue_buffering_drain(void)
135 {
136  output_queue_track_t *const track = &em_locm.output_queue_track;
137 
138  for (unsigned int i = 0; i < track->idx_cnt; i++) {
139  int qidx = track->idx[i];
140  queue_elem_t *output_q_elem = track->used_queues[qidx];
141  env_spinlock_t *lock = &output_q_elem->output.lock;
142 
143  /*
144  * drain if lock available, otherwise another core is already
145  * draining so no need to do anything.
146  */
147  if (env_spinlock_trylock(lock)) {
148  output_queue_drain(output_q_elem);
149  env_spinlock_unlock(lock);
150  }
151 
152  track->idx[i] = 0;
153  track->used_queues[qidx] = NULL;
154  }
155  track->idx_cnt = 0;
156 }
