95 /**
96  * Init values for the event-state counters 'free_cnt' and 'send_cnt'.
97  *
98  * The counters are 32-bit but are updated as one combined 64-bit atomic var,
99  * thus the init values are in the middle of the u32-range to avoid wraparounds
100  * when decrementing below '0'.
101  */
102 #define FREE_CNT_INIT ((uint16_t)0x0100) /* = 0 + 'offset' */
103 #define SEND_CNT_INIT ((uint16_t)0x0100) /* = 0 + 'offset' */
104 /** Initial event generation value */
105 #define EVGEN_INIT ((uint16_t)1)
106 /** Max evgen value before resetting to 'EVGEN_INIT' to avoid wrap */
107 #define EVGEN_MAX ((uint16_t)UINT16_MAX - 0x1000)
