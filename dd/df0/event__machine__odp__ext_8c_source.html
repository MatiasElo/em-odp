107 int pkt_enqueue(const odp_packet_t pkt_tbl[], const int num,
108  const em_queue_t queue)
109 {
110  if (unlikely(!pkt_tbl || num <= 0))
111  return 0;
112 
113  queue_elem_t *const q_elem = queue_elem_get(queue);
114  odp_event_t odp_event_tbl[num];
115  int sent = 0;
116 
117  odp_packet_to_event_multi(pkt_tbl, odp_event_tbl/*out*/, num);
118 
119  if (q_elem != NULL && q_elem->scheduled) {
120  /*
121  * Enqueue the events into a scheduled em-odp queue.
122  * No need to init the ev-hdrs - init is done in dispatch.
123  */
124  sent = odp_queue_enq_multi(q_elem->odp_queue,
125  odp_event_tbl, num);
126  } else {
127  em_event_t event_tbl[num];
128  event_hdr_t *evhdr_tbl[num];
129 
130  events_odp2em(odp_event_tbl, event_tbl/*out*/, num);
131 
132  /* Init the event-hdrs for incoming pkts */
133  pkt_evhdr_init_multi(pkt_tbl, event_tbl/*in/out*/,
134  evhdr_tbl/*out*/, num);
135 
136  if (q_elem == NULL) {
137  /* Send directly out via event chaining */
138  if (likely(queue_external(queue)))
139  sent = send_chaining_multi(event_tbl, evhdr_tbl,
140  num, queue);
141  } else if (q_elem->type == EM_QUEUE_TYPE_UNSCHEDULED) {
142  /* Enqueue into an unscheduled em-odp queue */
143  sent = odp_queue_enq_multi(q_elem->odp_queue,
144  odp_event_tbl, num);
145  } else if (q_elem->type == EM_QUEUE_TYPE_LOCAL) {
146  /* Send into an local em-odp queue */
147  sent = send_local_multi(event_tbl, evhdr_tbl,
148  num, q_elem);
149  } else if (q_elem->type == EM_QUEUE_TYPE_OUTPUT) {
150  /* Send directly out via an output em-odp queue */
151  sent = send_output_multi(event_tbl, evhdr_tbl,
152  num, q_elem);
153  }
154  }
155 
156  if (unlikely(sent < num)) {
157  sent = unlikely(sent < 0) ? 0 : sent;
158  odp_packet_free_multi(&pkt_tbl[sent], num - sent);
159  /*
160  * Event state checking: No need to adjust the event state
161  * since the events were never enqueued into EM.
162  */
163  }
164 
165  return sent;
166 }
