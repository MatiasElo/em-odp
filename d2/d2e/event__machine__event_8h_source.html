36 /**
37  * @file
38  * @defgroup em_event Events
39  * Operations on an event.
40  * @{
41  *
42  * All application processing is driven by events in the Event Machine. An event
43  * describes a piece of work. The structure of an event is implementation and
44  * event type specific: it may be a directly accessible buffer of memory, a
45  * descriptor containing a list of buffer pointers, a descriptor of a packet
46  * buffer etc.
47  *
48  * Applications use the event type to interpret the event structure.
49  *
50  * Events follow message passing semantics: an event has to be allocated using
51  * the provided API (em_alloc()) or received through queues by an EO callback
52  * function after which the event is owned by the application. Event ownership
53  * is transferred back to the system by using em_send() or em_free().
54  * An event not owned by the application should not be touched.
55  *
56  * Since em_event_t may not carry a direct pointer value to the event structure,
57  * em_event_pointer() must be used to translate an event to an event structure
58  * pointer (for maintaining portability).
59  *
60  * em_event_t is defined in event_machine_types.h
61  *
62  * @see em_event_pointer()
63  */

72 /**
73  * Allocate an event.
74  *
75  * The memory address of the allocated event is system specific and can depend
76  * on the given pool, event size and type. The returned event (handle) may refer
77  * to a memory buffer or a HW specific descriptor, i.e. the event structure is
78  * system specific.
79  *
80  * Use em_event_pointer() to convert an event (handle) to a pointer to an event
81  * structure.
82  *
83  * EM_EVENT_TYPE_SW with minor type '0' is reserved for direct portability -
84  * it is always guaranteed to produce an event with contiguous payload that can
85  * directly be used by the application up to the given size (no HW specific
86  * descriptors etc. are visible). This event payload will be 64-bit aligned
87  * by default (unless explicitly configured otherwise).
88  *
89  * EM_POOL_DEFAULT can be used as a pool handle if there's no need to use a
90  * specific event pool (up to the size- or event limits of that pool).
91  *
92  * Additionally it is guaranteed, that two separate buffers never share a cache
93  * line (to avoid false sharing).
94  *
95  * @param size Event size in octets (size > 0)
96  * @param type Event type to allocate
97  * @param pool Event pool handle
98  *
99  * @return The allocated event or EM_EVENT_UNDEF on error.
100  *
101  * @see em_free(), em_send(), em_event_pointer(), em_receive_func_t()
102  */
103 em_event_t em_alloc(size_t size, em_event_type_t type, em_pool_t pool);
104 
105 /**
106  * Allocate multiple events.
107  *
108  * Similar to em_alloc(), but allows allocation of multiple events, with same
109  * properties, with one function call.
110  * The em_alloc_multi() API function will try to allocate the requested number
111  * ('num') of events but may fail to do so, e.g. if the pool has run out of
112  * events, and will return the actual number of events that were successfully
113  * allocated from the given pool.
114  *
115  * @param[out] events Output event array, events are allocated and filled by
116  * em_alloc_multi(). The given array must fit 'num' events.
117  * @param num Number of events to allocate and write into 'events[]'
118  * @param size Event size in octets (size > 0)
119  * @param type Event type to allocate
120  * @param pool Event pool handle
121  *
122  * @return Number of events actually allocated from the pool (0 ... num) and
123  * written into the output array 'events[]'.
124  */
125 int em_alloc_multi(em_event_t events[/*out*/], int num,
126  size_t size, em_event_type_t type, em_pool_t pool);

128 /**
129  * Free an event.
130  *
131  * The em_free() function transfers ownership of the event back to the system
132  * and the application must not touch the event (or related memory buffers)
133  * after calling it.
134  *
135  * It is assumed that the implementation can detect the event pool that
136  * the event was originally allocated from.
137  *
138  * The application must only free events it owns. For example, the sender must
139  * not free an event after sending it.
140  *
141  * @param event Event to be freed
142  *
143  * @see em_alloc(), em_receive_func_t()
144  */
145 void em_free(em_event_t event);

158 /**
159  * Send an event to a queue.
160  *
161  * The event must have been allocated with em_alloc(), or received via an EO
162  * receive-function. The sender must not touch the event after calling em_send()
163  * as the ownership has been transferred to the system or possibly to the next
164  * receiver. If the return status is *not* EM_OK, the ownership has not been
165  * transferred and the application is still responsible for the event (e.g. may
166  * free it).
167  *
168  * EM does not currently define guaranteed event delivery, i.e. EM_OK return
169  * value only means the event was accepted for delivery. It could still be lost
170  * during delivery (e.g. due to a removed queue or system congestion, etc).
171  *
172  * @param event Event to be sent
173  * @param queue Destination queue
174  *
175  * @return EM_OK if successful (accepted for delivery).
176  *
177  * @see em_alloc()
178  */
179 em_status_t em_send(em_event_t event, em_queue_t queue);

182  * Send multiple events to a queue.
183  *

185  * performance gain.
186  * The function returns the number of events actually sent. A return value equal
187  * to the given 'num' means that all events were sent. A return value less than
188  * 'num' means that only the first 'num' events were sent and the rest must be
189  * handled by the application.
190  *
191  * @param events Array of events to send
192  * @param num Number of events.
193  * The array 'events[]' must contain 'num' entries.
194  * @param queue Destination queue
195  *
196  * @return number of events successfully sent (equal to num if all successful)

198  * @see em_send()
199  */

306 /**
307  * Mark the event as "sent".
308  *
309  * Indicates a user-given promise to EM that the event will later appear into
310  * 'queue' by some means other than an explicit user call to em_send...().
311  * Calling em_event_mark_send() transfers event ownership away from the user,
312  * and thus the event must not be used or touched by the user anymore (the only
313  * exception is (hw) error recovery where the "sent" state can be cancelled by
314  * using em_event_unmark_send() - dangerous!).
315  *
316  * Example use case:
317  * A user provided output-callback function associated with a queue of type
318  * 'EM_QUEUE_TYPE_OUTPUT' can use this API when configuring a HW-device to
319  * deliver the event back into EM. The HW will eventually "send" the event and
320  * it will "somehow" again appear into EM for the user to process.
321  *
322  * EM will, after this API-call, treat the event as "sent" and any further API
323  * operations or usage might lead to EM errors (depending on the error-check
324  * level), e.g. em_send/free/tmo_set/ack(event) etc. is forbidden after
325  * em_event_mark_send(event).
326  *
327  * @note Registered API-callback hooks for em_send...() (em_api_hook_send_t)
328  * will NOT be called.
329  * @note Marking an event "sent" with an event group (corresponding to
330  * em_send_group()) is currrently NOT supported.
331  *
332  * @param event Event to be marked as "sent"
333  * @param queue Destination queue (must be scheduled, i.e. atomic,
334  * parallel or parallel-ordered)
335  *
336  * @return EM_OK if successful
337  *
338  * @see em_send(), em_event_unmark_send()
339  */
340 em_status_t em_event_mark_send(em_event_t event, em_queue_t queue);
341 
342 /**
343  * Unmark an event previously marked as "sent" (i.e mark as "unsent")
344  *
345  * @note This is for recovery situations only and can potenially crash the
346  * application if used incorrectly!
347  *
348  * Revert an event's "sent" state, as set by em_event_mark_send(), back to the
349  * state before that function call.
350  * Any further usage of the event after em_event_mark_send(), by EM or
351  * the user, will result in error when calling em_event_unmark_send() since the
352  * state has become unrecoverable.
353  * => the only allowed EM API call after em_event_mark_send() is
354  * em_event_unmark_send() if it is certain that the event, due to some
355  * external error, will never be sent into EM again otherwise.
356  * Calling em_event_unmark_send() transfers event ownership back to the user
357  * again.
358  *
359  * @note This is the only valid case of using an event that the user no
360  * longer owns - all other such uses leads to fatal error.
361  *
362  * @code
363  * em_status_t err;
364  * hw_err_t hw_err;
365  *
366  * // 'event' owned by the user
367  * err = em_event_mark_send(event, queue);
368  * if (err != EM_OK)
369  * return err; // NOK
370  * // 'event' no longer owned by the user - don't touch!
371  *
372  * hw_err = config_hw_to_send_event(...hw-cfg..., event, queue);
373  * if (hw_err) {
374  * // hw config error - the event can be recovered if it is
375  * // certain that the hw won't send that same event.
376  * // note: the user doesn't own the event here and actually
377  * // uses an obsolete event handle to recover the event...
378  * err = em_event_unmark_send(event);
379  * if (err != EM_OK)
380  * return err; // NOK
381  * // 'event' recovered, again owned by the user
382  * em_free(event);
383  * }
384  * @endcode
385  *
386  * @param event Event previously marked as "sent" with em_event_mark_send(),
387  * any other case will be invalid!
388  *
389  * @return EM_OK if successful
390  *
391  * @see em_send(), em_event_mark_send()
392  */
393 em_status_t em_event_unmark_send(em_event_t event);
